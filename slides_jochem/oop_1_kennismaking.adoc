== Laten we een autorace game gaan maken.

[source,php]
----
<?php
$car_1 = 'Ford Mustang GT';
$pos_car_1 = 0;
$finish = 25;

while ($pos_car_1 != $finish) {
  $pos_car_1++;
}

echo $car_1 . " made it!";
----
