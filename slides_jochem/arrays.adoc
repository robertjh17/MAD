== Multidimensional array

== Multidimensional array

[source,php]
----
$my_list = [
  'groepen' => [1, 8, 2489],
  'aanduiding' => 'a'
  'woonplaats' => 'zwolle',
  'gemiddeld' => 3.14,
  'cijfers' => [
    'gemiddeld' => 3.14,
    'max' => 8,
    'min' => 1.3
  ]
];
----
