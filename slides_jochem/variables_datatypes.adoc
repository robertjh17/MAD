[source,php]
----
<pre>
<?php
print_r($_GET);

echo "\n\n\n"; // drie lege regels

foreach ($_GET as $key => $val) {
  echo "$key=$val\n"; // \n is het karakter voor een nieuwe regel
}
----

== for

[source,php]
----
<?php
$lijst = [1, 238, 43, 2536, 12, 238, 99];

for ($i=0; $i<count($lijst); $i++)
{
  echo $lijst[$i] . "\n";
}
?>
----
